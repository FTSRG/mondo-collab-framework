template triple-rule (gold, condition, correspondence, front, security-check, priority) {
	let gold-readable  = {gold; security-check[::READ]} in 
	let mapped = {gold-readable; condition; correspondence; front} in
		//	let mapped-putback = {gold;          condition; correspondence; front} in
		//	let mapped-get     = {gold-readable; condition; correspondence; front} in
	get { 
		rule get.add        (!mapped; condition; gold-readable)                   --> (create front; create correspondence;)    with +priority;
		rule get.remove     (!mapped; condition; front)                           --> (delete front; obsolete correspondence;)  with -priority;
	}
	putback {
		rule putback.add    (!mapped; condition; front) /* TODO if unreadable?*/  --> (create gold; create correspondence; if !LS(security-check[::WRITE]) rollback;)   with +priority;
		rule putback.remove (!mapped; condition; gold-readable)                   --> (if !LS(security-check[::WRITE]) rollback; delete gold; obsolete correspondence;) with -priority;
	}		
} // smallest priority comes first

// TODO assumption: obfuscated attributes are read-only
triple-rule attributes (priority: 50) gold {
	Gold$EObject.attributeSlot(goldEObject, eAttribute, goldValue);
} condition {
	Correspond$EObject(goldEObject, frontEObject);
} front {
	Front$EObject.attributeSlot(frontEObject, eAttribute, frontValue);
} security-check[::READ] {
	// TODO defaults?
	ReadJudgement$Attribute(goldEObject, eAttribute, <user>, Prevailing);
	frontValue == eval(
		switch(Prevailing) {
			case ::OBFUSCATE: obfuscate(goldValue)
			case ::PERMIT   : goldValue 
			case ::DENY     : null /* IncQuery treats this as a failed check */
		}
	);
} security-check[::WRITE] {
	// TODO defaults?
	WriteJudgement$Attribute(goldEObject, eAttribute, <user>, ::PERMIT);
}
